#####进行以下操作时需要系统可以连外网####


#系统安装好后进行两删除一清空
sed -ri '/HWADDR|UUID/d' /etc/sysconfig/network-scripts/ifcfg-eth*
>/etc/udev/rules.d/70-persistent-net.rules

#增加到
vim /etc/rc.local
>/etc/udev/rules.d/70-persistent-net.rules



#简单的系统优化
#修改yum源--centos6
yum -y install wget
wget -O /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-6.repo
wget -O /etc/yum.repos.d/epel.repo http://mirrors.aliyun.com/repo/epel-6.repo
yum clean all >/dev/null
yum makecache >/dev/null


#关闭iptables防火墙
/etc/init.d/iptables stop    #临时关闭
chkconfig iptables off		 #永久关闭
chkconfig |grep iptables

#关闭SELINUX
setenforce 0				#临时关闭
sed -i 's/SELINUX=enforcing/SELINUX=disabled/g' /etc/selinux/config		#永久关闭


#设置时间同步
echo "#times sync by willem at $(date +%F)" >>/var/spool/cron/root
echo "*/5 * * * * /usr/sbin/ntpdate time.nist.gov >/dev/null 2>&1" >> /var/spool/cron/root


#安装常用的工具
yum install gcc gcc-c++  wget cmake curl unzip zip telnet vim tree nmap sysstat lrzsz dos2unix -y

####以上优化做完后，就可以将新建的虚机关闭，然后进行克隆
